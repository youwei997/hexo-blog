## 函数形式
1. 指令函数触发时机：
   1. 指令与元素成功绑定时（一开始）
   2. 指令所在的模板被重新解析时（绑定的值变化时）
   

## 全局指令

-   使用：`v-has:键="'值'"` -> `v-has:add="'user-create'"`

```js
// 自定义指令
app.directive("has", {
    // 第一个是dom元素，第二个是包含指令键值的对象。arg是键（参数）类似v-on:click="function" 的click，value是值 类似v-on:click="function" 的function
    // v-has:键="'值'"。 值如果传字符串必须再包一层单引号，否则就是一个变量

    // beforeMount。如果是 beforeMount 就无法操作dom（增删改），因为dom还在内存中，还没渲染到页面上
    mounted: function (el, binding) {
        console.log(el, binding);
        // 获取按钮权限
        const actionList = storage.getItem("actionList");
        const value = binding.value;
        // 查找权限列表中是否有对应的权限标识
        const hasPermission = actionList.includes(value);
        if (!hasPermission) {
            // 隐藏按钮
            el.style = "display:none";
            // 删除按钮dom
            el.parentNode.removeChild(el);
            // setTimeout(() => el.parentNode.removeChild(el), 0) // 使用beforeMount时 ，使用宏任务形式，不是很了解，以后再看看
        }
    },
});
```

## 组件指令

```vue
<template>
	<div>
    	<input v-focus />
    </div>
</template>
<script>
    
    const directives = {
        focus:{
            mounted(el){
                el.focus();
            }
        }
    }
    
    export default {
        directives
    }
    
</script>
```

## 全局指令

```js
//main.js

// 这里等于下面的
// app.directive('focus',(el,binding)=>{
//  el.focus();
//})

app.directive('focus,'{
    mounted(el,binding){
    	el.focus();
	},
    updated(el,binding){
    	el.focus();
	}
 })
```

## 指令传值（value）

```vue
<template>
	<div>
    	<input v-top="100" />
    </div>
</template>
<script>
    
    const directives = {
        top:{
            // 第一个参数为dom，第二个为参数
            mounted(el,binding){
               el.style.top = (binding.value + 'px')
            },
            updated(el,binding){
               el.style.top = (binding.value + 'px')
            },
        }
    }
    
    export default {
        directives
    }
    
</script>
```

## 指令参数（arg）

```vue
<template>
	<div>
    	<input v-pos:top="100" />
    </div>
</template>
<script>
    
    const directives = {
        top:{
            // 第一个参数为dom，第二个为参数
            // binding： value 值，arg 参数
            mounted(el,binding){
               el.style[binding.arg] = (binding.value + 'px')
            },
            updated(el,binding){
               el.style[binding.arg] = (binding.value + 'px')
            },
        }
    }
    
    export default {
        directives
    }
    
</script>
```
