- 组件 （自定义组件）

```js
import React from "react";
import reactLogo from "../assets/react.svg";
class News extends React.Component {
  constructor(props) {
    // super必须写在其他this之前
    super(props);
    this.state = {
      inputValue: "",
      name: "李四",
      style: {
        color: "red",
        fontSize: "40px",
      },
      list: ["1", "2", "3"],
      list2: [<h1 key="1">h1标签1</h1>, <h1 key="2">h1标签2</h1>],
      inputRef: "",
    };
    this.run3 = this.run3.bind(this);
    this.inputRef = React.createRef();
  }
  run1() {
    console.log(this.state.name);
  }
  run2() {
    console.log(this.state.name);
  }
  run3() {
    console.log(this.state.name);
  }
  setData = () => {
    this.setState({
      name: "张三",
    });
  };
  setName(name) {
    this.setState({
      name: name,
    });
  }
  handleEvent(e) {
    console.log(e);
    e.target.style.background = "red";
    // 获取dom元素上的属性
    console.log(e.target.getAttribute("aid"));
  }
  inputChange = (e) => {
    this.setState({
      inputValue: e.target.value,
    });
  };
  getInputValue() {
    console.log(this.state.inputValue);
  }
  getInputRef = () => {
    console.log(this.inputRef.current);
  };
  inputKeyUp = (e) => {
    console.log(e.code);
  };
  render() {
    return (
      <div>
        <h1>{this.state.name}</h1>
        <ul>
          <li>列表</li>
        </ul>
        {/* 绑定state里的属性 */}
        <h1>绑定state里的属性</h1>
        <div style={this.state.style}>绑定state里的属性</div>
        {/* 引入图片 */}
        <h1>引入图片</h1>
        <img src={reactLogo} alt="" />
        <img
          src="https://www.baidu.com/img/PCtm_d9c8750bed0b3c7d089fa7d55720d6cf.png"
          alt=""
        />
        {/* 循环数组 */}
        <h1>循环数组</h1>
        {this.state.list.map((item) => {
          return <div key={item}>{item}</div>;
        })}
        {this.state.list2}
        {/* 改变this */}
        <h1>改变this</h1>
        <button onClick={this.run1.bind(this)}>改变this的方法1</button>
        <br />
        <button
          onClick={() => {
            this.run2();
          }}
        >
          改变this的方法2
        </button>
        <br />
        <button onClick={this.run3}>改变this的方法3</button>
        <br />
        <button onClick={this.run3}>改变this的方法3</button>
        <br />
        <button onClick={this.setData}>设置state里的值</button>
        <br />

        <button
          onClick={() => {
            this.setName("二狗");
          }}
        >
          方法传值
        </button>
        {/* 事件对象 */}
        <h1>事件对象</h1>
        <button aid="123" onClick={this.handleEvent.bind(this)}>
          事件对象
        </button>
        <br />
        <input onChange={this.inputChange}></input>
        <button onClick={this.getInputValue.bind(this)}>提交</button>
        <br />
        {/* 获取ref dom */}
        <h1>获取ref </h1>
        <input ref={this.inputRef}></input>
        <button onClick={this.getInputRef}>获取ref</button>
        <br />
        {/* 键盘事件 */}
        <h1>键盘事件</h1>
        <input onKeyUp={this.inputKeyUp}></input>
      </div>
    );
  }
}

export default News;
```
